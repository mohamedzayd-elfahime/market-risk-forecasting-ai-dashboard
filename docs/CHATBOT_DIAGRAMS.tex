\section*{2. Vector RAG Index Construction}

{\small
The fixed chatbot knowledge base is built offline from curated Markdown files. The documents are cleaned, split into overlapping chunks, embedded with a lightweight sentence-transformer, and persisted in a local Chroma vector database.
}

\vspace{2mm}

\begin{center}
\resizebox{0.96\textwidth}{!}{%
\begin{tikzpicture}[node distance=15mm and 18mm]

  \node[neutral, wideblock] (cmd) {Command\\\texttt{python -m}\\\texttt{backend.chatbot.rag.build\_index}};
  \node[neutral, right=of cmd, wideblock] (docs) {Markdown documents\\\texttt{backend/chatbot/rag/docs}};
  \node[app, right=of docs] (clean) {Markdown\\cleaning};
  \node[app, right=of clean] (headers) {Header split\\\#, \#\#, \#\#\#};

  \node[app, below=22mm of docs] (chunks) {Child chunks\\200 tokens\\overlap 40};
  \node[app, right=of chunks] (refine) {Long chunk\\refinement};
  \node[control, right=of refine, wideblock] (embed) {Embedding model\\\texttt{all-MiniLM-L6-v2}};
  \node[control, right=of embed] (vectors) {Normalized\\vectors};
  \node[db, right=of vectors] (db) {Local\\Chroma DB};

  \draw[arrow] (cmd) -- (docs);
  \draw[arrow] (docs) -- (clean);
  \draw[arrow] (clean) -- (headers);
  \draw[arrow] (headers) |- (chunks);
  \draw[arrow] (chunks) -- (refine);
  \draw[arrow] (refine) -- (embed);
  \draw[arrow] (embed) -- (vectors);
  \draw[arrow] (vectors) -- (db);

\end{tikzpicture}%
}
\end{center}

\section*{3. Runtime RAG Retrieval}

{\small
At runtime, the intent router decides whether document retrieval is useful. If RAG is required, the user question is embedded and matched against the local Chroma database. Retrieved passages are then injected into the controlled prompt.
}

\vspace{2mm}

\begin{center}
\resizebox{0.96\textwidth}{!}{%
\begin{tikzpicture}[node distance=12mm and 13mm]

  \node[neutral] (question) {User\\question};
  \node[control, right=of question] (intent) {Detected\\intent};
  \node[decision, right=of intent] (route) {RAG\\needed?};

  \node[control, above right=11mm and 14mm of route] (embedq) {Question\\embedding};
  \node[safety, below right=11mm and 14mm of route] (norag) {No RAG\\numeric context};

  \node[db, right=of embedq] (db) {Chroma\\\texttt{vector\_db}};
  \node[rag, right=of db] (search) {Similarity search\\top-k passages};
  \node[rag, right=of search] (format) {Passage formatting\\source + section};

  \node[app, below=19mm of format] (prompt) {Final prompt};
  \node[app, right=of prompt] (llm) {Local LLM};

  \node[rag, below=of prompt] (numeric) {Forecast\\Backtest\\Metadata};

\section*{4. Intent Routing}

{\small
The intent router uses lightweight embeddings rather than a fully generative model. User questions are compared with intent centroids built from annotated examples, then routed to the minimal context required for the answer.
}

\vspace{2mm}

\begin{center}
\resizebox{0.92\textwidth}{!}{%
\begin{tikzpicture}[node distance=14mm and 18mm]

  \node[neutral, wideblock] (question) {User question};
  \node[control, right=of question, wideblock] (embedding) {Light embedding\\\texttt{all-MiniLM-L6-v2}};
  \node[control, right=of embedding, wideblock] (centroids) {Intent centroids\\annotated examples};
  \node[control, right=of centroids] (intent) {Cosine similarity\\intent router};

  \node[control, below left=16mm and 43mm of intent] (help) {Help request};
  \node[control, below left=16mm and 15mm of intent] (def) {Definition query};

\section*{5. Response Guardrails}

{\small
The chatbot remains generative, but its output is checked before it reaches the dashboard. Guardrails detect unsupported numbers, investment advice, VaR/ES confusion, and critical metric interpretation errors, then apply a targeted correction when needed.
}

\vspace{2mm}

\begin{center}
\resizebox{0.92\textwidth}{!}{%
\begin{tikzpicture}[node distance=13mm and 15mm]

  \node[app] (raw) {Raw LLM\\answer};
  \node[app, right=of raw] (repair) {Answer\\repair};
  \node[control, right=of repair] (validate) {Final\\validation};

  \node[decision, above right=11mm and 14mm of validate] (numbers) {Unsupported\\numbers?};
  \node[decision, right=30mm of validate] (advice) {Investment\\advice?};
  \node[decision, below right=11mm and 14mm of validate] (vares) {VaR / ES\\confusion?};

  \node[safety, right=18mm of numbers] (fallback) {Targeted correction\\or fallback};
  \node[safety, right=18mm of advice] (refusal) {Controlled\\refusal};
  \node[safety, right=18mm of vares] (explain) {Definition\\correction};

  \node[app, right=22mm of refusal] (final) {Final API\\answer};
  \node[app, below=14mm of final] (payload) {API payload};

  \draw[arrow] (raw) -- (repair);
  \draw[arrow] (repair) -- (validate);

  \draw[arrow] (validate) -- (numbers);
  \draw[arrow] (validate) -- (advice);
  \draw[arrow] (validate) -- (vares);

  \draw[arrow] (numbers) -- node[labeltxt, above] {yes} (fallback);
  \draw[arrow] (advice) -- node[labeltxt, above] {yes} (refusal);
  \draw[arrow] (vares) -- node[labeltxt, above] {yes} (explain);

  \draw[arrow] (fallback) -- (final);
  \draw[arrow] (refusal) -- (final);
  \draw[arrow] (explain) -- (final);

  \draw[dashedarrow]
    (validate.south) -- ++(0,-16mm) -| node[labeltxt, near start, below] {ok} (final.south);

  \draw[arrow] (final) -- (payload);

\end{tikzpicture}%
}
\end{center}
